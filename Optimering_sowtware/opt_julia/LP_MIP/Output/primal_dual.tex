\section*{Primal problem}
\subsection*{Problemformulering}
\[
\begin{aligned}
\max  \;& 4 x_1  +3 x_2  +5 x_3
\\ \text{u.b.b. } \;& 
2 x_1  +x_2  +3 x_3 \le 4
\\ &  -x_1  +x_2 \ge 0
\\ & x_1  +x_3 = 1
\\ \text{Samt fortegnskrav: } \;& x_1 \in \mathbb{R}, x_2 \geq 0, x_3 \geq 0
\end{aligned}
\]
\section*{Dual problem}
\subsection*{Problemformulering}
\[
\begin{aligned}
\min  \;& 4 y_1  +y_3
\\ \text{u.b.b. } \;& 
2 y_1  -y_2  +y_3 = 4
\\ & y_1  +y_2 \ge 3
\\ & 3 y_1  +y_3 \ge 5
\\ \text{Samt fortegnskrav: } \;& y_1 \geq 0, y_2 \leq 0, y_3 \in \mathbb{R}
\end{aligned}
\]
